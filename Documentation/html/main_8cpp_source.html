167bool initWiFi()
168{
169 if (ssid == "" || ip == "")
170 {
171 Serial.println("Undefined SSID or IP address.");
172 return false;
173 }
174
175 WiFi.mode(WIFI_STA);
176 localIP.fromString(ip.c_str());
177 localGateway.fromString(gateway.c_str());
178
179 if (!WiFi.config(localIP, localGateway, subnet, dns))
180 {
181 Serial.println("STA Failed to configure");
182 return false;
183 }
184 WiFi.begin(ssid.c_str(), pass.c_str());
185 Serial.println("Connecting to WiFi...");
186
187 unsigned long currentMillis = millis();
188 previousMillis = currentMillis;
189
190 while (WiFi.status() != WL_CONNECTED)
191 {
192 currentMillis = millis();
193 if (currentMillis - previousMillis >= interval)
194 {
195 Serial.println("Failed to connect.");
196 return false;
197 }
198 }
199
200 Serial.println(WiFi.localIP());
201 return true;
202}
